    <div class="setup">
      <label for="maxNumbers">Número de bolas: </label>
      <select id="maxNumbers">
        <option value="30">30</option>
        <option value="60">60</option>
        <option value="75">75</option>
        <option value="90" selected>90</option>
      </select>
      <button class="btn-primary" onclick="startBingo()">Empezar</button>
    </div>

    <div class="current-number" id="currentNumber">--</div>

    <div class="controls">
      <button class="btn-success" onclick="nextBall()">Siguiente bola</button>
      <button class="btn-primary" onclick="startAuto()">Automático</button>
      <button class="btn-warning" onclick="pauseAuto()">Pausa</button>
      <button class="btn-danger" onclick="resetBingo()">Reiniciar</button>
    </div>

    <div class="numbers" id="numbers"></div>
  </div>

  <script>
    let max = 90;
    let availableNumbers = [];
    let drawnNumbers = [];
    let autoInterval = null;

    function startBingo() {
      max = parseInt(document.getElementById('maxNumbers').value);
      resetBingo();
      availableNumbers = Array.from({ length: max }, (_, i) => i + 1);
      renderNumbers();
    }

    function nextBall() {
      if (availableNumbers.length === 0) return;
      const index = Math.floor(Math.random() * availableNumbers.length);
      const number = availableNumbers.splice(index, 1)[0];
      drawnNumbers.push(number);
      document.getElementById('currentNumber').innerText = number;
      updateDrawn(number);
    }

    function startAuto() {
      if (autoInterval) return;
      autoInterval = setInterval(nextBall, 7000);
    }

    function pauseAuto() {
      clearInterval(autoInterval);
      autoInterval = null;
    }

    function resetBingo() {
      pauseAuto();
      availableNumbers = [];
      drawnNumbers = [];
      document.getElementById('currentNumber').innerText = '--';
      document.getElementById('numbers').innerHTML = '';
    }

    function renderNumbers() {
      const container = document.getElementById('numbers');
      container.innerHTML = '';
      for (let i = 1; i <= max; i++) {
        const div = document.createElement('div');
        div.className = 'number';
        div.id = 'num-' + i;
        div.innerText = i;
        container.appendChild(div);
      }
    }

    function updateDrawn(number) {
      const el = document.getElementById('num-' + number);
      if (el) el.classList.add('drawn');
    }
  </script>
